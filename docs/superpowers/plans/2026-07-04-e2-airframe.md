```python
def auw_g(components):
    """Повна злітна маса, г."""
    return sum(c["mass_g"] for c in components)

def center_of_mass_mm(components):
    """Центр мас (x,y,z) у мм від центру рами (зважене середнє)."""
    m = auw_g(components)
    return tuple(sum(c["mass_g"] * c[ax] for c in components) / m for ax in ("x_mm", "y_mm", "z_mm"))
```

```python
import math

G = 9.80665

def thrust_total_g(thrust_per_motor_g, n=4):
    return thrust_per_motor_g * n

def twr(thrust_total_g, auw_g):
    return thrust_total_g / auw_g

def pitch_speed_ms(kv, v_batt, pitch_in, eta_rpm=0.8):
    """Теоретична швидкість за кроком пропа (без опору): RPM*pitch."""
    rpm = kv * v_batt * eta_rpm
    return rpm * (pitch_in * 0.0254) / 60.0

def drag_limited_speed_ms(thrust_max_n, weight_n, cd, area_m2, rho=1.225):
    """Макс. горизонтальна швидкість: forward-складова тяги = лобовий опір.
    forward = sqrt(T^2 - W^2) (вертикаль тримає вагу)."""
    if thrust_max_n <= weight_n:
        return 0.0
    forward = math.sqrt(thrust_max_n ** 2 - weight_n ** 2)
    return math.sqrt(2 * forward / (rho * cd * area_m2))

def top_speed_ms(kv, v_batt, pitch_in, thrust_max_n, weight_n, cd, area_m2, eta_rpm=0.8, rho=1.225):
    return min(pitch_speed_ms(kv, v_batt, pitch_in, eta_rpm),
               drag_limited_speed_ms(thrust_max_n, weight_n, cd, area_m2, rho))

def climb_rate_ms(thrust_max_n, weight_n, cd, area_top_m2, rho=1.225):
    """Стала вертикальна швидкість: T = W + опір(v)."""
    if thrust_max_n <= weight_n:
        return 0.0
    return math.sqrt(2 * (thrust_max_n - weight_n) / (rho * cd * area_top_m2))

def flight_time_min(cap_ah, v_nom, dod, eta, avg_power_w):
    energy_wh = cap_ah * v_nom * dod * eta
    return energy_wh / avg_power_w * 60.0
```

```python
import math

def kinetic_energy_j(mass_kg, v_ms):
    return 0.5 * mass_kg * v_ms ** 2

def momentum_kgms(mass_kg, v_ms):
    return mass_kg * v_ms

def arm_max_speed_ms(arm_w_mm, arm_h_mm, arm_len_mm, arm_cd, sigma_allow_mpa, rho=1.225):
    """Безпечна макс. швидкість за згином PETG-променя (консоль, розподілений аеро-навант.).
    q = 0.5*rho*v^2*cd*h [Н/м]; M_root = q*L^2/2; sigma = M*(h/2)/I; I = w*h^3/12.
    Розв'язок відносно v при sigma = sigma_allow."""
    w = arm_w_mm / 1000.0
    h = arm_h_mm / 1000.0
    L = arm_len_mm / 1000.0
    sigma = sigma_allow_mpa * 1e6
    I = w * h ** 3 / 12.0
    c = h / 2.0
    # sigma = (0.5*rho*v^2*cd*h) * L^2/2 * c / I  ->  v^2 = sigma*I / (c * 0.25*rho*cd*h*L^2)
    denom = c * 0.25 * rho * arm_cd * h * L ** 2
    return math.sqrt(sigma * I / denom)
```

```python
"""ТТХ-звіт E2 з єдиного конфіга + CLI."""
from cuas.design import e2_config as cfg
from cuas.design.mass import auw_g, center_of_mass_mm
from cuas.design.propulsion import (thrust_total_g, twr, top_speed_ms,
                                     climb_rate_ms, flight_time_min, G)
from cuas.design.ram import kinetic_energy_j, arm_max_speed_ms

def build_report():
    auw = auw_g(cfg.COMPONENTS)
    com = center_of_mass_mm(cfg.COMPONENTS)
    thrust_g = thrust_total_g(cfg.THRUST_PER_MOTOR_G)
    thrust_n = thrust_g / 1000.0 * G
    weight_n = auw / 1000.0 * G
    vmax = top_speed_ms(cfg.MOTOR_KV, cfg.V_BATT_FULL, cfg.PROP_PITCH_IN,
                        thrust_n, weight_n, cfg.CD_BODY, cfg.FRONTAL_AREA_M2)
    climb = climb_rate_ms(thrust_n, weight_n, cfg.CD_BODY, cfg.TOP_AREA_M2)
    ft = flight_time_min(cfg.BATT_CAP_AH, cfg.V_BATT_NOMINAL, cfg.BATT_DOD, cfg.BATT_ETA, cfg.EST_AVG_POWER_W)
    ke = kinetic_energy_j(auw / 1000.0, vmax)
    arm_cap = arm_max_speed_ms(cfg.ARM_W_MM, cfg.ARM_H_MM, cfg.ARM_LEN_MM, cfg.ARM_CD, cfg.PETG_SIGMA_ALLOW_MPA)
    return {
        "auw_g": round(auw, 1),
        "com_mm": tuple(round(v, 1) for v in com),
        "twr": round(twr(thrust_g, auw), 2),
        "top_speed_kmh": round(vmax * 3.6, 1),
        "climb_ms": round(climb, 1),
        "flight_time_min": round(ft, 1),
        "ram_ke_j": round(ke, 0),
        "arm_cap_kmh": round(arm_cap * 3.6, 1),
    }

def _print(r):
    print("=== E2 5\" auto-ram — TTX ===")
    for k, v in r.items():
        print(f"{k:16}: {v}")

if __name__ == "__main__":
    _print(build_report())
```
